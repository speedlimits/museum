#
# Content Distribution Configuration
#

cache=(
  0 = Memory(
    policy = LRU(size = 200M)
  )
  1 = Network(
    services = (mhash:/// = file:///./Staging)
  )
  2 = Disk(
    directory = Cache
    policy = LRU(size = 3000M)
  )
  3 = Network(
    services = $download
  )
)

namelookup=(
    meru:/// = (
      0 = http://localhost:8000/names
      1 = http://www.sirikata.com/namesbackup/2009-09-09.06
      2 = http://graphics.stanford.edu/~danielrh/dns/names/global
    )
    meru://cplatz@/ = (
       1 = http://graphics.stanford.edu/~danielrh/dns/names/global/cplatz
       2 = file:///./Staging/names-cplatz.txt
    )
    meru://daniel@/ = (
       1 = http://graphics.stanford.edu/~danielrh/dns/names/global/daniel
    )
    meru://meru@/ = http://graphics.stanford.edu/~danielrh/dns/names/global/meru
)

download=(
    mhash:/// = (
       0 = file:///./Cache
       1 = http://www.sirikata.com/content/assets
       2 = http://graphics.stanford.edu/~danielrh/uploadsystem/files/global
       3 = file:///./Staging
    )
)

nameupload=(
    meru:/// = file:///./Staging/names.txt
)

upload=(
    mhash:// = file:///./Staging
)
